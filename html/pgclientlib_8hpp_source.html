156  void connect_tcp(const std::string host = "localhost",
157  const std::string service = "postgresql")
158  {

173  throw std::runtime_error("Could not open socket");

504  void process_reply(const server_message_header& msg)
505  {
506  switch(msg.code)
507  {
508  case 'A': // NotificationResponse
509  {
510  auto buf = read_remaining(msg);
511  parse_notifications(buf);
512  break;
513  }
514  case 'C': // CommandComplete
515  {
516  auto buf = read_remaining(msg);
517  notifications.push(buf2str(buf));
518  state = session_state::complete;
519  break;
520  }
521  case 'c': // CopyDone
522  {
523  auto buf = read_remaining(msg);
524  state = session_state::copy_done;
525  break;
526  }
527  case 'D': // DataRow
528  {
529  row_queue.push(read_remaining(msg));
530  break;
531  }
532  case 'd': // CopyData
533  {
534  row_queue.push(read_remaining(msg));
535  break;
536  }
537  case 'E': // ErrorResponse
538  {
539  auto buf = read_remaining(msg);
540  parse_notifications(buf);
541  break;
542  }
543  case 'G': // CopyInResponse
544  {
545  auto buf = read_remaining(msg);
546  if (buf[0]) buf_fmt = buffer_format::copy_binary;
547  else buf_fmt = buffer_format::copy_text;
548  state = session_state::copy_in;
549  break;
550  }
551  case 'H': // CopyOutResponse
552  {
553  auto buf = read_remaining(msg);
554  if (buf[0]) buf_fmt = buffer_format::copy_binary;
555  else buf_fmt = buffer_format::copy_text;
556  state = session_state::copy_out;
557  clear_row_queue();
558  break;
559  }
560  case 'I': // EmptyQuery
561  {
562  read_remaining(msg);
563  notifications.push("[Empty request]");
564  break;
565  }
566  case 'K': // BackendKeyData
567  {
568  pid = read<boost::endian::big_int32_t>();
569  skey = read<boost::endian::big_int32_t>();
570  break;
571  }
572  case 'N': // NoticeResponse
573  {
574  auto buf = read_remaining(msg);
575  parse_notifications(buf);
576  break;
577  }
578  case 'R': // Authentication
579  {
580  auto auth_code = read<boost::endian::big_int32_t>();
581  if (auth_code)
582  throw std::runtime_error("Autentication mode not supported");
583  auto auth_msg = get_reply();
584  process_reply(auth_msg);
585  if (is_error(auth_msg))
586  {
587  throw std::runtime_error("Error in startup; cannot continue");
588  }
589  handle_replies();
590  break;
591  }
592  case 'S': // ParameterStatus
593  {
594  parse_params(read_remaining(msg));
595  break;
596  }
597  case 'T': // RowDescription
598  {
599  field_map.clear();
600  auto buf = read_remaining(msg);
601  boost::endian::big_int16_t nfields;
602  std::memcpy(&nfields, &buf[0], 2);
603  auto start_pos = buf.begin() + 2;
604  while (nfields--)
605  {
606  field_descriptor fd;
607  auto first_null = std::find(start_pos, buf.end(), '\0');
608  std::string field_name(start_pos, first_null);
609  std::memcpy(&fd, &*first_null + 1, sizeof(fd));
610  field_map.push_back(std::make_pair(field_name, fd));
611  start_pos = first_null + sizeof(fd) + 1;
612  }
613  buf_fmt = buffer_format::query;
614  clear_row_queue();
615  break;
616  }
617  case 'Z': // ReadyForQuery
618  {
619  switch (read<std::uint8_t>())
620  {
621  case 'I': ts = transaction_status::idle; break;
622  case 'T': ts = transaction_status::active; break;
623  case 'E': ts = transaction_status::error; break;
624  default: throw std::runtime_error("Invalid transaction status");
625  }
626  state = session_state::ready_for_query;
627  if (echo_codes) std::cout << std::endl;
628  break;
629  }
630  default:
631  {
632  read_remaining(msg);
633  std::stringstream ss;
634  ss << "Cannot handle server message with code '" << msg.code << "'";
635  throw std::runtime_error(ss.str());
636  break;
637  }
638  } // switch msg.code
639  } // process reply

718  row_type row_to_strings(const buffer_type& rr) const
719  {
720  switch (buf_fmt)
721  {
722  case buffer_format::query:
723  {
724  row_type res;
725  assert(rr.size() > 2);
726  boost::endian::big_int16_t n;
727  std::memcpy(&n, &rr[0], 2);
728  auto i = &rr[2];
729  for (int j = 0; j != n; ++j)
730  {
731  boost::endian::big_int32_t sz;
732  std::memcpy(&sz, i, 4); i += 4;
733  if (sz < 0)
734  {
735  res.emplace_back();
736  continue;
737  }
738  if (field_map.at(j).second.frmt_code)
739  {
740  std::stringstream ss;
741  while (sz--)
742  {
743  if (std::isprint(*i)) ss << *i;
744  else ss << '.';
745  ++i;
746  }
747  res.emplace_back(ss.str());
748  continue;
749  }
750  res.emplace_back(i, i + sz);
751  i += sz;
752  }
753  return res;
754  }
755  case buffer_format::copy_text:
756  {
757  return row_type({{std::string(rr.begin(), rr.end())}});
758  }
759  case buffer_format::copy_binary:
760  {
761  std::string res(rr.size(), '\0');
762  std::transform(std::begin(rr), std::end(rr),
763  std::begin(res), [](std::uint8_t x)
764  {
765  return std::isprint(x) ? x : '.';
766  });
767  return row_type({res});
768  }
769  default: throw std::runtime_error("Unknown buffer format");
770  }
771  }
